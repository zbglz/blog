### 响应式原理部分代码实现

现在我们不使用vue.js,看看响应式实现的原理。

```js vue2
<div id="app">

  <div>
    <input type="text" v-model="msg" />
  </div>
  
  独立文本节点：{{msg}}
  
  <div>
    标签节点下的文本节点：{{msg}}
  </div>
  
  <div>
    <span>标签节点嵌套下的文本节点：{{msg}}</span>
  </div>
  
</div>

<script>
  class Vue {
    constructor(options) {
      // 1、保存数据
      this.$options = options
      this.$data = options.data
      this.$el = options.el
      
      // 2、将data数据添加到响应式系统中
      new Observer(this.$data)
      
      // 3、代理this.$data的数据
      Object.keys(this.$data).forEach(key => {
        this._proxy(key)
      })
      
      // 4、处理el
      new Compiler(this.$el, this)
    }
    
    _proxy(key) {
      Object.defineProperty(this, key, {
        configurable: true,
        enumerable: true,
        set(newValue){
          this.$data[key] = newValue
        },
        get(){
          return this.$data[key]
        }
      })
    }
  }
  
  class Observer {
    constructor(data) {
      this.data = data
      Object.keys(data).forEach(key => {
        this.defineReactive(this.data, key, data[key])
      })
    }
    
    defineReactive(data, key, val) {
      //一个属性key，对应一个Dep对象
      const dep = new Dep()
      Object.defineProperty(data, key, {
        enumerable: true,
        configurable: true,
        get(){
          if(Dep.target){
            dep.addSub(Dep.target)
          }
          return val
        },
        set(newValue){
          if(newValue === val){
            return
          }
          val = newValue
          dep.notify()
        }
      })
    }
  }
  
  class Dep {
    constructor(){
      this.subs = []
    }
    
    addSub(sub){
      this.subs.push(sub)
    }
    
    notify(){
      this.subs.forEach(sub => {
        sub.update()
      })
    }
  }
  
  class Watcher {
    constructor(node, name, vm) {
      this.node = node
      this.name = name
      this.vm = vm
      Dep.target = this
      this.update()
      Dep.target = null
    }
    
    update() {
      // this.node.nodeValue = this.vm[this.name]
      // update重写
      switch(this.node.nodeName){
        case "#text":
          // 文本节点除了语法绑定的值，还有其他文本的情况。
          if(this.node.oldValue === undefined){
            this.node.oldValue = this.node.nodeValue
          }
          var reg = new RegExp( "{{" + this.name + "}}", "g" )
          this.node.nodeValue = this.node.oldValue.replace(reg, this.vm[this.name])
          break
        case "INPUT":
          this.node.value = this.vm[this.name]
          // 设置input的默认value值。
          var attr = document.createAttribute("value")
          attr.value = this.vm[this.name]
          this.node.setAttributeNode(attr)
          break
      }
    }
  }
  
  const reg = /\{\{(.+)\}\}/
  class Compiler {
    //这个el就是#app
    constructor(el, vm) {
      this.el = document.querySelector(el)
      this.vm = vm
      
      this.frag = this._createFragment()
      this.el.appendChild(this.frag)
    }
    
    _createFragment() {
      const frag = document.createDocumentFragment()
      
      let child
      while (child = this.el.firstChild){
        this._compile(child)  //解析<div>{{message}}</div>
        frag.appendChild(child)
      }
      return frag
    }
    
    _compile(node) {
      if(node.nodeType === 1) { // 标签节点
        /* const attrs = node.attributes
        if(attrs.hasOwnProperty('v-model')) { 
          const name = attrs['v-model'].nodeValue
          node.addEventListener('input', e => {
            this.vm[name] = e.target.value
          })
        } */
        // 标签节点重写
        const attrs = node.attributes
        if(attrs.hasOwnProperty('v-model')) { // v-model双向绑定
          const name = attrs['v-model'].nodeValue
          new Watcher(node, name, this.vm)
          node.addEventListener('input', e => {
            this.vm[name] = e.target.value
          })
        }else{ 
          // 其他标签节点下的值绑定。
          // 教程中，如果用标签将input或者{{}}包裹，会出错，这里需要判断标签嵌套的情况。
          for(var i=0;i<node.childNodes.length;i++){
            this._compile(node.childNodes[i])
          }
        }
      }
      if(node.nodeType === 3) { // 文本节点
        if(reg.test(node.nodeValue)) {
          const name = RegExp.$1.trim()
          new Watcher(node, name, this.vm)
        }
        
      }
    }
  }
  
  const app = new Vue({
    el: "#app",
    data: {
      msg: 'hello'
    }
  })
  console.log("仅支持：")
  console.log("1、mustache语法值绑定，一个标签内只能绑定一个值")
  console.log("2、v-model双向绑定。你可以在此处尝试修改app.msg的值")
</script>
```
