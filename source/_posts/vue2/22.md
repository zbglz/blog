### 全局组件

通过Vue注册的组件就是全局组件。


```js vue2
<script src="link/vue.min.2.6.14.js"></script>
<div id="app">
  <my-cpn></my-cpn>      
</div>
<script>
  const cpnC = Vue.extend({
    template:`
      <div>
        <h4>我是标题</h4>
        <p>我是内容</p>
      </div>
             `
  })
  
  // 通过Vue注册的组件就是全局组件。
  Vue.component('my-cpn',cpnC)
  
  const app = new Vue({
    el: "#app",
    data: {
    }
  })
</script>
```


全局组件在任何Vue实例中都能使用。


```js vue2
<script src="link/vue.min.2.6.14.js"></script>
<div id="app">
  我是Vue实例app,挂载的元素
  <my-cpn></my-cpn>
</div>
<hr />
<div id="app1">
  我是Vue实例app1,挂载的元素
  <my-cpn></my-cpn>
</div>

<script>
  const cpnC = Vue.extend({
    template:`
      <div>
        <h4>我是标题</h4>
        <p>我是内容</p>
      </div>
             `
  })
  
  // 全局注册的组件在任何Vue实例中都能使用。
  Vue.component('my-cpn',cpnC)
  
  const app = new Vue({
    el: "#app",
    data: {
    }
  })
  //这里只做演示, 一般在实际项目中, 只会有一个Vue实例。
  const app1 = new Vue({
    el: "#app1",
    data: {
    }
  })
</script>
```

### 局部组件

在vue实例内部注册的组件, 或在组件内注册的组件就是局部组件。**局部组件**只会在**注册它的Vue实例**, **或组件**内有效。


```js vue2
<script src="link/vue.min.2.6.14.js"></script>

<div id="app">
  Vue实例app注册了myCpn组件
  <my-cpn></my-cpn>
</div>
<hr />
<div id="app1">
  Vue实例app1没有注册myCpn组件
  <my-cpn></my-cpn>
</div>

<script>
  const cpnC = Vue.extend({
    template:`
      <div>
        <h4>我是标题</h4>
        <p>我是内容</p>
      </div>
             `
  })
  
  const app = new Vue({
    el: "#app",
    data: {
    },
    components:{
      myCpn:cpnC // 在components属性中, 进行局部注册。
      
      //语法： 使用时的标签名:构造器。
      //如果设置使用时的标签名与构造器名称一样, 则可以直接简写为 cpnC
      //但是注意：标签名中有大写时, 在使用时需要转换 如：myCpn应该写成my-cpn标签。
    }
  })

  const app1 = new Vue({
    el: "#app1",
    data: {
    }
  })
</script>
```
